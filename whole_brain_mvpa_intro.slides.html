Intro: Deep neural network (DNN)¶

DNN is a neural network based computational model with multiple non-linear transformation layers.¶

model parameters (i.e., weight parameters) can be determined based on iterative learning to minimize a cost function (e.g., an error between true target values and estimaetd target values).

Q: How we can solve this problem?¶

Let's start with a very simple linear model as follows:¶

$$ y = w x + b,$$

where $x$ and $y$ are the input and output data, $w$ and $b$ are the weight and bias parameters.

The trainable parameters, $w$ and $b$ can be obtained by minimizing a cost function (J) such as the mean-squared error (MSE) between an actual target ($y$) and a predicted target ($\hat{y}$) using training/validation data: $$ J = \sum_i |y_i - \hat{y}_i|^2 $$

*The code snippet to define the linear model, to train the model, and to predict the model is shown below:

# linear regression model using TF
class TFlinear(object):
    def __init__(self, x_dim, learning_rate = 0.01):
        self.x_dim = x_dim
        self.learning_rate = learning_rate
        self.g = tf.Graph()
        # build the model
        with self.g.as_default():
            self.build()
            self.init_op = tf.global_variables_initializer()
    
    def build(self):
        self.X = tf.placeholder(dtype=tf.float32, 
                    shape=(None, self.x_dim), name='x_input')
        self.y = tf.placeholder(dtype=tf.float32, 
                    shape=(None), name='y_output')
        print(self.X)
        print(self.y)
        w = tf.Variable(tf.truncated_normal(shape=(1,1), stddev=0.1), name='weight')
        b = tf.Variable(tf.ones(shape=(1))*0.1, name='bias')
        print(w)
        print(b)

        self.z_net = tf.squeeze(self.X*w + b, name='z_net')
        print(self.z_net)

        self.y_hat = self.z_net
        sqr_err = tf.square(self.y - self.y_hat, name='sqr_err')
        print(sqr_err)

        self.mse_cost = tf.reduce_mean(sqr_err, name='mse_cost')
        print(self.mse_cost)

        self.cost = self.mse_cost

        optimizer = tf.train.GradientDescentOptimizer(
            learning_rate = self.learning_rate, name='GradientDescent')

        self.train_op = optimizer.minimize(self.cost, name='train_op')

# This function is to train the linear model
def train_model(sess, model, X_train, y_train, n_epochs=100):
    sess.run(model.init_op)
    tr_cost = []
    for i in range(n_epochs):
        # This is for linear model for now
        _, cost = sess.run([model.train_op, model.cost], feed_dict={model.X: X_train, model.y: y_train})
        tr_cost.append(cost),

        if not i % 100:
            print('Epoch {:3d}: {:.4f}'.format(i, cost))
    return tr_cost # for one layer (linear) model

# function to return the predicted output 
def predict_model(sess, model, X_test, y_test):
    y_pred = sess.run(model.y_hat, feed_dict={model.X: X_test})
    mse_err = np.mean(np.square(y_test - y_pred))
    return y_pred, mse_err

Non-linear neural network (NN) model¶

Now, let's consider a NN model with one hidden layer¶

In this model, the hidden node representation is as follows:

$$ z_i = w_{1,i} x + b_{1,i}, \hspace{1cm} h_i = \varphi (z_i), \hspace{1cm} y = \sum_{j=1}^{n} w_{2,j} h_j + b_2, $$

where $\varphi(\cdot)$ is a non-linear function (e.g., sigmoid, tanh, and ReLU).

*Code snippet is below.

class TF2layer(object):
    def __init__(self, x_dim, h_dim, l1param=0.001, l2param=0.00001, learning_rate = 0.01):
        self.x_dim = x_dim
        self.h_dim = h_dim
        self.l1param = l1param # L1-norm parameter
        self.l2param = l2param # L2-norm parameter
        self.learning_rate = learning_rate
        self.g = tf.Graph()
        # build the model
        with self.g.as_default():
            self.build()
            self.init_op = tf.global_variables_initializer()
In [ ]:
    def build(self):
        self.X = tf.placeholder(dtype=tf.float32, 
                    shape=(None, self.x_dim), name='x_input')
        self.y = tf.placeholder(dtype=tf.float32, 
                    shape=(None), name='y_output')
        print(self.X)
        print(self.y)
        self.w1 = tf.Variable(tf.truncated_normal(shape=(self.x_dim, self.h_dim), stddev=0.1), name='weight1')
        self.b1 = tf.Variable(tf.ones(shape=(self.h_dim))*0.1, name='bias1')
        print(self.w1)
        print(self.b1)

        self.z_net = tf.squeeze(tf.matmul(self.X, self.w1) + self.b1, name='z_net')
        print(self.z_net)

        #self.h = self.z_net
        self.h = tf.tanh(self.z_net)

        self.w2 = tf.Variable(tf.truncated_normal(shape=(self.h_dim, 1), stddev=0.1), name='weight2')
        self.b2 = tf.Variable(tf.ones(shape=(1))*0.1, name='bias2')
        print(self.w2)
        print(self.b2)

        self.y_hat = tf.squeeze(tf.matmul(self.h, self.w2) + self.b2, name='y_hat')
        print(self.y_hat)

        sqr_err = tf.square(self.y - self.y_hat, name='sqr_err')
        print(sqr_err)

        self.l1norm = tf.multiply(self.l1param, tf.add(tf.reduce_sum(tf.abs(self.w1)), tf.reduce_sum(tf.abs(self.w2))))
        self.l2norm = tf.multiply(self.l2param, tf.add(tf.reduce_sum(tf.pow(self.w1, 2)), tf.reduce_sum(tf.pow(self.w2, 2))))

        self.mse_cost = tf.reduce_mean(sqr_err, name='mse_cost')
        print(self.mse_cost)

        self.cost = self.mse_cost

        optimizer = tf.train.GradientDescentOptimizer(
            learning_rate = self.learning_rate, name='GradientDescent')

        self.train_op = optimizer.minimize(self.mse_cost, name='train_op')

# This function is to train the two-layer model
def train_model(sess, model, X_train, y_train, n_epochs=100):
    sess.run(model.init_op)
    tr_cost, tr_mse_cost, tr_l1norm, tr_l2norm, = [], [], [], []
    for i in range(n_epochs):
        # The below is for two-layer model
        _, mse_cost, l1norm, l2norm, cost = sess.run([model.train_op, model.mse_cost, model.l1norm
                    , model.l2norm, model.cost], feed_dict={model.X: X_train, model.y: y_train})
        tr_cost.append(cost), tr_mse_cost.append(mse_cost), tr_l1norm.append(l1norm), tr_l2norm.append(l2norm)

        if not i % 100:
            print('Epoch {:3d}: {:.4f}'.format(i, cost))
    return tr_cost, tr_mse_cost, tr_l1norm, tr_l2norm # for two layer model

# function to return the predicted output

    y_pred = sess.run(model.y_hat, feed_dict={model.X: X_test})

As a summary (for now):¶

1. As the number of hidden nodes increased too much, the model seemed significantly overfitted to some training samples.¶

2. Thus, the generalization performance was substantially compromised.¶

3. We can also expect this potential overfitting issue from whole brain MVPA¶

i.e., number of input voxels is often greater than 50,000 and thus, number of hidden nodes need to be increased.
On the other hand, the number of training samples are limited.

Solution: let's regularize the model weights¶

Goal is to penalize the model weights and in turn to reduce the model complexity.¶

$$ J = \alpha \sum_i |y_i - \hat{y}_i|^2 + \beta \sum_j |w_{1,j}| + \gamma \sum_k || w_{2,k} ||^2, $$

where $\alpha$, $\beta$, and $\gamma$ are the learning rate for the sum of squarred error, L1-norm, and L2-norm of weights, respectively.

With regularization of weights, one hidden layer neural net model¶

Number of hidden nodes = 5,000¶

$\beta = 1.0$ and $\gamma = 1.0$¶

# Hoyer's sparseness
def hsp(data_vec):
    data_vec_dim = np.max(data_vec.shape)
    denom = np.sqrt(data_vec_dim) - 1
    numer = np.sqrt(data_vec_dim) - \
        np.sum(np.abs(data_vec))/np.sqrt(np.sum(np.square(data_vec)))
    hsp_val = np.divide(numer, denom)
    return hsp_val

# Non-zero ratio (NZR)
def nzr(data_vec, epsilon=1e-2):
    # since the scale matters, first, normalize, then evaluate the non-zero ratio
    data_vec_norm = np.divide(data_vec, np.sqrt(np.sum(np.square(data_vec))))
    nzr_val = np.sum(np.abs(data_vec_norm) > epsilon) / np.max(data_vec.shape)
    return nzr_val

Ok, now you saw how the L1-norm and L2-norm regularization to weights change the capacity of the networks.¶

Particularly, the L1-norm regularization of weights controls the degree of feature selection.¶

The problem is that a fixed value of L1-norm parameter keep increasing the level of sparsity:¶

Also, for a fixed overall learning rate ($\alpha$), the sparseness reached differently depending on the fixed L1-norm parameter.
For a fixed L1-norm parameter, the sparseness of weights are very different depending on an overall learning rate.

One possible scheme to optimize the model and consequently to select the optimal set of features (particularly for the whole brain MVPA) is to find an optimal level of weight sparsity.¶

Find an optimal level of weight sparsity using an adaptive L1-norm parameter

$$ \beta (t+1) = \beta (t) - \mu (\rho - nzr(W(t))), $$

where $\beta$ is a L1-norm regularization parameter, $\rho$ is a target weight sparsity level, $W(t)$ is a model weight, $\mu$ is a learning rate adaptively to change the L1-norm regularization parameter, and $t$ is a iteration number of neural network training.

Thus, an optimal target sparsity level can be found using a cross-validation with training and validation data
e.g., grid search among candidate target non-zero ratio values: {0.2, 0.5, 0.8}

Few more remarks: Sparsity metric¶

Non-zero ratio (NZR) is based on L0-norm¶

$$ NZR = \frac{N(|W| > \epsilon)}{N(W)}, $$

where $N(\cdot)$ is a function to count the number of elements and $\epsilon$ is a threshold to define a zero/non-zero value.

Hoyer's sparseness (HSP) is based on L1-norm and L2-norm¶

$$ HSP = \frac{\sqrt{N} - \frac{||W||_1}{||W||_2}}{\sqrt{N} - 1} $$
